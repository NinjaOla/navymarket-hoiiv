# Test 1: Scripted localisation + meta_effect — THE WORKING SOLUTION
# Chain: set_variable sale_type_id → scripted loc get_sale_ship_type → meta_effect transfer_ship
# Scripted loc maps numeric id to loc key (ship_token_*) which resolves to raw sub-unit name
# This eliminates the need for if-chains in transfer logic
country_event = {
    id = shiptest.1
    title = "Test: Scripted Loc + meta_effect transfer"
    desc = "The confirmed working pattern for dynamic ship transfers"
    is_triggered_only = yes

    option = {
        name = "carrier (sale_type_id=1)"
        set_variable = { sale_type_id = 1 }
        log = "shiptest.1: type=[Root.get_sale_ship_type], var=[?sale_type_id]"
        meta_effect = {
            text = {
                transfer_ship = { type = [T] target = NOR }
            }
            T = "[Root.get_sale_ship_type]"
            debug = yes
        }
    }
    option = {
        name = "Cancels"
    }
}
