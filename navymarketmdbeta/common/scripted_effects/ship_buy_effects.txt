### Buyer-side scripted effects for Ship Market

# Finds all countries with selling_ships flag, populates seller_list array
populate_seller_slots = {
	clear_array = seller_list
	every_country = {
		limit = {
			has_country_flag = selling_ships
			NOT = { tag = ROOT }
		}
		ROOT = {
			add_to_array = { seller_list = PREV }
		}
	}
}

# Copies selected seller's listing variables to buyer scope for GUI display
load_seller_listings = {
	# Populate 1-element array for seller name display (change_scope gridbox)
	clear_array = selected_seller_list
	add_to_array = { selected_seller_list = selected_seller }
	var:selected_seller = {
		ROOT = {
			set_variable = { buy_avail_carrier = PREV.ships_selling_carrier }
			set_variable = { buy_avail_destroyer = PREV.ships_selling_destroyer }
			set_variable = { buy_avail_frigate = PREV.ships_selling_frigate }
			set_variable = { buy_avail_corvette = PREV.ships_selling_corvette }
			set_variable = { buy_avail_cruiser = PREV.ships_selling_cruiser }
			set_variable = { buy_avail_attack_submarine = PREV.ships_selling_attack_submarine }
			set_variable = { buy_avail_missile_submarine = PREV.ships_selling_missile_submarine }
			set_variable = { buy_avail_helicopter_operator = PREV.ships_selling_helicopter_operator }
			set_variable = { buy_avail_battleship = PREV.ships_selling_battleship }
			set_variable = { buy_avail_battle_cruiser = PREV.ships_selling_battle_cruiser }
			set_variable = { buy_avail_stealth_destroyer = PREV.ships_selling_stealth_destroyer }
			set_variable = { buy_avail_stealth_frigate = PREV.ships_selling_stealth_frigate }
			set_variable = { buy_avail_stealth_corvette = PREV.ships_selling_stealth_corvette }
		}
	}
}

# Executes ship transfer from seller to buyer using meta_effect for dynamic type
execute_ship_purchase = {
	set_variable = { buy_transfer_count = buy_quantity }
	meta_effect = {
		text = {
			# Transfer ships one at a time
			while_loop_effect = {
				limit = { check_variable = { buy_transfer_count > 0 } }
				var:selected_seller = {
					transfer_ship = { type = [T] target = ROOT prefer_name = "SALE" }
				}
				subtract_from_variable = { buy_transfer_count = 1 }
			}
			# Subtract purchased quantity from seller's listing
			var:selected_seller = {
				subtract_from_variable = { ships_selling_[T] = ROOT.buy_quantity }
				# Clear variable if listing is depleted
				if = {
					limit = {
						NOT = { check_variable = { ships_selling_[T] > 0 } }
					}
					clear_variable = ships_selling_[T]
				}
				# Clear selling flag if no listings remain
				if = {
					limit = {
						NOT = {
							OR = {
								check_variable = { ships_selling_carrier > 0 }
								check_variable = { ships_selling_destroyer > 0 }
								check_variable = { ships_selling_frigate > 0 }
								check_variable = { ships_selling_corvette > 0 }
								check_variable = { ships_selling_cruiser > 0 }
								check_variable = { ships_selling_attack_submarine > 0 }
								check_variable = { ships_selling_missile_submarine > 0 }
								check_variable = { ships_selling_helicopter_operator > 0 }
								check_variable = { ships_selling_battleship > 0 }
								check_variable = { ships_selling_battle_cruiser > 0 }
								check_variable = { ships_selling_stealth_destroyer > 0 }
								check_variable = { ships_selling_stealth_frigate > 0 }
								check_variable = { ships_selling_stealth_corvette > 0 }
							}
						}
					}
					clr_country_flag = selling_ships
				}
			}
			# Update buyer's local availability display
			subtract_from_variable = { buy_avail_[T] = buy_quantity }
		}
		T = "[Root.get_buy_ship_type]"
	}
	log = "Ship market: [Root.GetTag] bought [?buy_quantity] [Root.get_buy_ship_type] from [var:selected_seller.GetTag]"
}
